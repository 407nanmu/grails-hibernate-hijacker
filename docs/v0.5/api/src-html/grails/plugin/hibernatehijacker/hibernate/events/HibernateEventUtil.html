001    package grails.plugin.hibernatehijacker.hibernate.events;
002    
003    import java.lang.reflect.Array;
004    import java.lang.reflect.Method;
005    import java.util.ArrayList;
006    import java.util.List;
007    
008    import org.hibernate.cfg.Configuration;
009    import org.hibernate.event.EventListeners;
010    
011    /**
012     * 
013     * @see http://www.docjar.com/html/api/org/hibernate/event/EventListeners.java.html
014     * @author Kim A. Betti
015     */
016    public class HibernateEventUtil {
017    
018        /**
019         * Finds all *EventListener interfaces registered with the class
020         * and adds them as listeners.
021         * @param configuration
022         * @param listener
023         * @throws Exception
024         */
025        public static void addListener(EventListeners eventListeners, Object listener) {
026            try {
027                List<Class<?>> listenerInterfaces = findEventListenerInterfaces(listener.getClass());
028                for (Class<?> listenerInterface : listenerInterfaces) {
029                    Object[] existingListeners = getExistingListeners(eventListeners, listenerInterface);
030                    Object[] newListeners = addToExistingListeners(existingListeners, listenerInterface, listener);
031                    setListeners(eventListeners, listenerInterface, newListeners);
032                }
033            } catch (Exception ex) {
034                String exMessage = "Unable to add event listeners from instance " + listener;
035                throw new RuntimeException(exMessage, ex);
036            }
037        }
038    
039        private static List<Class<?>> findEventListenerInterfaces(Class<?> listenerClass) {
040            List<Class<?>> listenerInterfaces = new ArrayList<Class<?>>();
041            findEventListenerInterfaces(listenerClass, listenerInterfaces);
042            return listenerInterfaces;
043        }
044    
045        private static void findEventListenerInterfaces(Class<?> listenerClass, List<Class<?>> listenerInterfaces) {
046            for (Class<?> listenerInterface : listenerClass.getInterfaces()) {
047                String canonicalName = listenerInterface.getCanonicalName();
048                if (canonicalName.startsWith("org.hibernate.event.") && canonicalName.endsWith("EventListener")) {
049                    listenerInterfaces.add(listenerInterface);
050                }
051            }
052    
053            Class<?> parentClass = listenerClass.getSuperclass();
054            if (parentClass != null) {
055                findEventListenerInterfaces(parentClass, listenerInterfaces);
056            }
057        }
058    
059        private static void setListeners(EventListeners eventListeners, Class<?> listenerInterface, Object[] newListeners) throws Exception {
060            for (Method method : eventListeners.getClass().getMethods()) {
061                if (method.getName().startsWith("set")) {
062                    Class<?>[] paramTypes = method.getParameterTypes();
063                    if (paramTypes.length == 1 && paramTypes[0].getComponentType() == listenerInterface) {
064                        method.invoke(eventListeners, new Object[] { newListeners });
065                        return;
066                    }
067                }
068            }
069    
070            throw new Exception("Unable to set listeners for " + listenerInterface);
071        }
072    
073        public static void addListener(Configuration configuration, String type, Object listener) {
074            EventListeners listeners = configuration.getEventListeners();
075            Class<?> listenerClass = listeners.getListenerClassFor(type);
076            addListener(configuration, type, listenerClass, listener);
077        }
078    
079        @SuppressWarnings("unchecked")
080        private static <T> void addListener(Configuration configuration, String type, Class<T> listenerClass, Object listener) {
081            EventListeners eventListeners = configuration.getEventListeners();
082    
083            try {
084                T[] existingListeners = getExistingListeners(eventListeners, listenerClass);
085                T[] newListeners = (T[]) addToExistingListeners(existingListeners, listenerClass, listener);
086                configuration.setListeners(type, newListeners);
087            } catch (Exception ex) {
088                String message = "Unable to add Hibernate event listener: " + listener + " for type: " + listenerClass.getSimpleName();
089                throw new RuntimeException(message, ex);
090            }
091        }
092    
093        private static Object[] addToExistingListeners(Object[] existingListeners, Class<?> listenerInterface, Object newListener) {
094            Object[] newListeners = (Object[]) Array.newInstance(listenerInterface, existingListeners.length + 1);
095            System.arraycopy(existingListeners, 0, newListeners, 0, existingListeners.length);
096            newListeners[existingListeners.length] = newListener;
097            return newListeners;
098        }
099    
100        @SuppressWarnings("unchecked")
101        private static <T> T[] getExistingListeners(EventListeners eventListeners, Class<T> type) throws Exception {
102            for (Method method : eventListeners.getClass().getMethods()) {
103                if (method.getName().startsWith("get")) {
104                    Class<?> returnType = method.getReturnType();
105                    if (returnType.isArray() && returnType.getComponentType() == type) {
106                        return (T[]) method.invoke(eventListeners);
107                    }
108                }
109            }
110    
111            throw new Exception("Unable to get existing listeners for " + type.getSimpleName());
112        }
113    
114    
115    }
