001    package grails.plugin.hibernatehijacker.hibernate.events;
002    
003    import grails.plugin.hibernatehijacker.hibernate.HibernateConfigPostProcessor;
004    
005    import org.hibernate.HibernateException;
006    import org.hibernate.cfg.Configuration;
007    import org.hibernate.event.EventListeners;
008    
009    /**
010     * Responsible for adding Hibernate event listeners provided by the plugin.
011     * Currently we just add one event listener, although this listener is implementing
012     * most of the Hibernate event listener interfaces in order to translate them into Hawk events.
013     * @author Kim A. Betti
014     */
015    public class EventListenerConfigurator implements HibernateConfigPostProcessor {
016    
017        private HibernateEventListener hibernateEventListener;
018    
019        @Override
020        public void doPostProcessing(final Configuration configuration) throws HibernateException {
021            registerEventlisteners(configuration);
022        }
023    
024        private void registerEventlisteners(Configuration configuration) {
025            EventListeners eventListeners = configuration.getEventListeners();
026            HibernateEventUtil.addListener(eventListeners, hibernateEventListener);
027        }
028    
029        public void setHibernateEventListener(HibernateEventListener hibernateEventListener) {
030            this.hibernateEventListener = hibernateEventListener;
031        }
032    
033    }
