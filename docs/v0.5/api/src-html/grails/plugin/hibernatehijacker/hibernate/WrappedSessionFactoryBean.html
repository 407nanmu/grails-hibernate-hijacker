001    package grails.plugin.hibernatehijacker.hibernate;
002    
003    import grails.plugin.hibernatehijacker.hibernate.events.HibernateEventUtil;
004    
005    import java.util.ArrayList;
006    import java.util.List;
007    import java.util.Map;
008    
009    import org.apache.commons.logging.Log;
010    import org.apache.commons.logging.LogFactory;
011    import org.codehaus.groovy.grails.orm.hibernate.ConfigurableLocalSessionFactoryBean;
012    import org.codehaus.groovy.grails.orm.hibernate.HibernateEventListeners;
013    import org.hibernate.HibernateException;
014    import org.hibernate.SessionFactory;
015    import org.hibernate.cfg.Configuration;
016    import org.hibernate.context.CurrentSessionContext;
017    import org.springframework.orm.hibernate3.SpringSessionContext;
018    
019    /**
020     * The SessionFactory is still built the usual way, but instead of returning the
021     * actual instance we're creating and returning a proxy.
022     * 
023     * Other features like event listeners can be enabled by injecting
024     * HibernateConfigurationPostProcessor beans.
025     * 
026     * @author Kim A. Betti <kim.betti@gmail.com>
027     */
028    public class WrappedSessionFactoryBean extends ConfigurableLocalSessionFactoryBean {
029    
030        private static final Log log = LogFactory.getLog(WrappedSessionFactoryBean.class);
031    
032        private SessionFactoryProxyFactory sessionFactoryProxyFactory;
033        private List<HibernateConfigPostProcessor> hibernateConfigPostProcessors = new ArrayList<HibernateConfigPostProcessor>();
034        private Map<String, Object> listenerMap;
035    
036        /**
037         * If no plugins has specified another CurrentSessionContextClass (like the
038         * webflow plugin) we'll use SpringSessionContext.
039         */
040        private Class<? extends CurrentSessionContext> currentSessionContextClass = SpringSessionContext.class;
041    
042        @Override
043        protected SessionFactory buildSessionFactory() throws Exception {
044            setExposeTransactionAwareSessionFactory(false);
045            SessionFactory realSessionFactory = super.buildSessionFactory();
046            return sessionFactoryProxyFactory.createSessionFactoryProxy(realSessionFactory, currentSessionContextClass);
047        }
048    
049        @Override
050        protected void postProcessConfiguration(final Configuration config) throws HibernateException {
051            for (HibernateConfigPostProcessor processor : hibernateConfigPostProcessors) {
052                log.debug("Passing Hibernate configuration to: " + processor.getClass().getSimpleName());
053                processor.doPostProcessing(config);
054            }
055    
056            super.postProcessConfiguration(config);
057            addListenersFromMap(config);
058        }
059    
060        /**
061         * Important: Do not call super!
062         * 
063         * @see http://jira.codehaus.org/browse/GRAILS-7211
064         * @see http://jira.codehaus.org/browse/GRAILS-5725
065         * @param listeners
066         */
067        @Override
068        public void setHibernateEventListeners(final HibernateEventListeners listeners) {
069            if (listeners != null) {
070                this.listenerMap = listeners.getListenerMap();
071            }
072        }
073    
074        private void addListenersFromMap(Configuration configuration) {
075            if (this.listenerMap != null) {
076                for (String type : this.listenerMap.keySet()) {
077                    Object listener = listenerMap.get(type);
078                    log.debug("Adding listener for " + type + ": " + listener.getClass().getName());
079                    System.out.println("Adding listener for " + type + ": " + listener.getClass().getName());
080                    HibernateEventUtil.addListener(configuration, type, listener);
081                }
082            }
083        }
084    
085        /**
086         * Make sure not to pass the argument to
087         * ConfigurableLocalSessionFactoryBean. We don't want Hibernate to make an
088         * instance of the CurrentSessionContext.
089         * 
090         * By listening in on this we're able to support plugins like webflow
091         * without introducing a compile time dependency.
092         */
093        @Override
094        @SuppressWarnings("unchecked")
095        public void setCurrentSessionContextClass(Class<?> currentSessionContextClass) {
096            this.currentSessionContextClass = (Class<? extends CurrentSessionContext>) currentSessionContextClass;
097        }
098    
099        public void setSessionFactoryProxyFactory(SessionFactoryProxyFactory sessionFactoryProxyFactory) {
100            this.sessionFactoryProxyFactory = sessionFactoryProxyFactory;
101        }
102    
103        public void setHibernateConfigPostProcessors(List<HibernateConfigPostProcessor> configPostProcessors) {
104            this.hibernateConfigPostProcessors = configPostProcessors;
105        }
106    
107    }
