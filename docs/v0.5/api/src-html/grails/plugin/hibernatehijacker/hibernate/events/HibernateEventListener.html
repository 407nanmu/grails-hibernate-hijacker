001    package grails.plugin.hibernatehijacker.hibernate.events;
002    
003    import grails.plugins.hawkeventing.EventBroker;
004    import grails.util.GrailsNameUtils;
005    
006    import java.util.Map;
007    import java.util.Set;
008    
009    import org.hibernate.HibernateException;
010    import org.hibernate.event.*;
011    
012    /**
013     * Brute force approach. 
014     * 
015     * This could probably be implemented in a more dynamic 
016     * fashion using Groovy, but with more overhead. 
017     * 
018     * @author Kim A. Betti
019     */
020    @SuppressWarnings("serial")
021    public class HibernateEventListener implements AutoFlushEventListener, DeleteEventListener, DirtyCheckEventListener, 
022            EvictEventListener, FlushEventListener, FlushEntityEventListener, LoadEventListener,
023            InitializeCollectionEventListener, LockEventListener, MergeEventListener, PersistEventListener, 
024            PostDeleteEventListener, PostInsertEventListener, PostLoadEventListener, PostUpdateEventListener, 
025            PreDeleteEventListener, PreInsertEventListener, PreLoadEventListener, PreUpdateEventListener, 
026            RefreshEventListener, ReplicateEventListener, SaveOrUpdateEventListener {
027    
028        
029        private EventBroker eventBroker;
030            
031        public void setEventBroker(EventBroker eventBroker) {
032            this.eventBroker = eventBroker;
033        }
034    
035        @Override
036        public void onSaveOrUpdate(SaveOrUpdateEvent event) throws HibernateException {
037            publishEvent("saveOrUpdate", event, event.getEntity());
038        }
039    
040        @Override
041        public void onReplicate(ReplicateEvent event) throws HibernateException {
042            publishEvent("replicate", event);
043        }
044        
045        @Override
046        @SuppressWarnings("rawtypes")
047        public void onRefresh(RefreshEvent event, Map arg1) throws HibernateException {
048            onRefresh(event);
049        }
050    
051        @Override
052        public void onRefresh(RefreshEvent event) throws HibernateException {
053            publishEvent("refresh", event);
054        }
055    
056        @Override
057        public boolean onPreUpdate(PreUpdateEvent event) {
058            publishEvent("preUpdate", event, event.getEntity());
059            return false;
060        }
061    
062        @Override
063        public void onPreLoad(PreLoadEvent event) {
064            publishEvent("preLoad", event, event.getEntity());
065        }
066    
067        @Override
068        public boolean onPreInsert(PreInsertEvent event) {
069            publishEvent("preInsert", event, event.getEntity());
070            return false;
071        }
072    
073        @Override
074        public boolean onPreDelete(PreDeleteEvent event) {
075            publishEvent("preDelete", event, event.getEntity());
076            return false;
077        }
078    
079        @Override
080        public void onPostUpdate(PostUpdateEvent event) {
081            publishEvent("postUpdate", event, event.getEntity());
082        }
083    
084        @Override
085        public void onPostLoad(PostLoadEvent event) {
086            publishEvent("postLoad", event, event.getEntity());
087        }
088    
089        @Override
090        public void onPostInsert(PostInsertEvent event) {
091            publishEvent("postInsert", event, event.getEntity());
092        }
093    
094        @Override
095        public void onPostDelete(PostDeleteEvent event) {
096            publishEvent("postDelete", event, event.getEntity());
097        }
098        
099        @Override
100        @SuppressWarnings("rawtypes")
101        public void onPersist(PersistEvent event, Map arg1) throws HibernateException {
102            onPersist(event);
103        }
104    
105        @Override
106        public void onPersist(PersistEvent event) throws HibernateException {
107            publishEvent("persist", event);
108        }
109    
110        @Override
111        @SuppressWarnings("rawtypes")
112        public void onMerge(MergeEvent event, Map arg1) throws HibernateException {
113            onMerge(event);
114        }
115        
116        @Override
117        public void onMerge(MergeEvent event) throws HibernateException {
118            publishEvent("merge", event, event.getEntity());
119        }
120    
121        @Override
122        public void onLock(LockEvent event) throws HibernateException {
123            publishEvent("lock", event);
124        }
125    
126        @Override
127        public void onInitializeCollection(InitializeCollectionEvent event) throws HibernateException {
128            publishEvent("initializeCollection", event);
129        }
130    
131        @Override
132        public void onLoad(LoadEvent event, LoadType arg1) throws HibernateException {
133            publishEvent("load", event);
134        }
135    
136        @Override
137        public void onFlushEntity(FlushEntityEvent event) throws HibernateException {
138            publishEvent("flushEntity", event, event.getEntity());
139        }
140    
141        @Override
142        public void onFlush(FlushEvent event) throws HibernateException {
143            publishEvent("flush", event);
144        }
145    
146        @Override
147        public void onEvict(EvictEvent event) throws HibernateException {
148            publishEvent("evict", event);
149        }
150    
151        @Override
152        public void onDirtyCheck(DirtyCheckEvent event) throws HibernateException {
153            publishEvent("dirtyCheck", event);
154        }
155        
156        @Override
157        @SuppressWarnings("rawtypes")
158        public void onDelete(DeleteEvent event, Set arg1) throws HibernateException {
159            onDelete(event);
160        }
161    
162        @Override
163        public void onDelete(DeleteEvent event) throws HibernateException {
164            publishEvent("delete", event);
165        }
166    
167        @Override
168        public void onAutoFlush(AutoFlushEvent event) throws HibernateException {
169            publishEvent("autoFlush", event);
170        }
171        
172        private void publishEvent(String eventName, AbstractEvent event) {
173            publishEvent(eventName, event, null);
174        }
175        
176        private void publishEvent(String eventName, AbstractEvent event, Object entity) {
177            StringBuilder fullEventName = new StringBuilder(100);
178            fullEventName.append("hibernate.").append(eventName);
179            
180            if (entity != null) {
181                final String entityName = GrailsNameUtils.getPropertyName(entity.getClass());
182                fullEventName.append(".").append(entityName);
183            }
184                
185            eventBroker.publish(fullEventName.toString(), event);
186        }
187        
188    }
