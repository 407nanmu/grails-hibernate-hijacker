001    package grails.plugin.hibernatehijacker.hibernate;
002    
003    import grails.plugins.hawkeventing.EventBroker;
004    import java.lang.reflect.InvocationHandler;
005    import java.lang.reflect.Method;
006    import org.hibernate.SessionFactory;
007    import org.hibernate.context.CurrentSessionContext;
008    
009    /**
010     * SessionFactory method invocation will be passing through here. 
011     * @author Kim A. Betti <kim.betti@gmail.com>
012     */
013    public class SessionFactoryInvocationHandler implements InvocationHandler {
014        
015        public static final String OPEN_SESSION = "openSession";
016        public static final String GET_CURRENT_SESSION = "getCurrentSession";
017        public static final String INTERCEPTED_SESSION_EVENT_NAME = "hibernate.sessionCreated";
018        
019        private SessionFactory realSessionFactory;
020        private EventBroker eventBroker;
021        private CurrentSessionContext defaultCurrentSessionContext;
022    
023        public SessionFactoryInvocationHandler(SessionFactory realSessionFactory, EventBroker eventBroker) {
024            this.realSessionFactory = realSessionFactory;
025            this.eventBroker = eventBroker;
026        }
027    
028        @Override
029        public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
030            final String methodName = method.getName();
031            
032            if (methodName.equals(GET_CURRENT_SESSION)) 
033                return defaultCurrentSessionContext.currentSession();
034            
035            Object returnedInstance = method.invoke(realSessionFactory, args);
036            if (methodName.equals(OPEN_SESSION)) 
037                eventBroker.publish(INTERCEPTED_SESSION_EVENT_NAME, returnedInstance);
038            
039            return returnedInstance;
040        }
041        
042        public void setDefaultCurrentSessionContext(CurrentSessionContext csc) {
043            this.defaultCurrentSessionContext = csc;
044        }
045        
046    }
