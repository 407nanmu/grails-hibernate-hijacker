001    package grails.plugin.hibernatehijacker.spring;
002    
003    import grails.plugin.hibernatehijacker.hibernate.HibernateConfigPostProcessor;
004    import grails.plugin.hibernatehijacker.hibernate.WrappedSessionFactoryBean;
005    
006    import java.util.Collection;
007    
008    import org.springframework.beans.BeansException;
009    import org.springframework.beans.MutablePropertyValues;
010    import org.springframework.beans.PropertyValue;
011    import org.springframework.beans.factory.config.BeanDefinition;
012    import org.springframework.beans.factory.config.BeanFactoryPostProcessor;
013    import org.springframework.beans.factory.config.ConfigurableListableBeanFactory;
014    import org.springframework.beans.factory.config.RuntimeBeanReference;
015    import org.springframework.core.Ordered;
016    
017    /**
018     * Replaces the default ConfigurableLocalSessionFactoryBean with
019     * WrappedSessionFactoryBean. It will also make sure that our replacement is
020     * wired with the required dependencies.
021     * 
022     * @author Kim A. Betti <kim.betti@gmail.com>
023     */
024    public class SessionFactoryPostProcessor implements BeanFactoryPostProcessor, Ordered {
025    
026        @Override
027        public void postProcessBeanFactory(ConfigurableListableBeanFactory beanFactory) throws BeansException {
028            BeanDefinition beanDef = beanFactory.getBeanDefinition("sessionFactory");
029            beanDef.setBeanClassName(WrappedSessionFactoryBean.class.getName());
030            setBeanProperties(beanFactory, beanDef);
031        }
032    
033        private void setBeanProperties(ConfigurableListableBeanFactory beanFactory, BeanDefinition beanDef) {
034            MutablePropertyValues properties = beanDef.getPropertyValues();
035            setHibernateProxyFactoryReference(properties);
036            setHibernateConfigPostProcessors(beanFactory, properties);
037        }
038    
039        private void setHibernateProxyFactoryReference(MutablePropertyValues properties) {
040            RuntimeBeanReference reference = new RuntimeBeanReference("sessionFactoryProxyFactory");
041            properties.add("sessionFactoryProxyFactory", reference);
042        }
043    
044        /**
045         * Looks up all beans implementing HibernateConfigPostProcessor and makes
046         * sure that they're injected into WrappedSessionFactoryBean
047         * 
048         * @param beanFactory
049         * @param properties
050         */
051        private void setHibernateConfigPostProcessors(ConfigurableListableBeanFactory beanFactory, MutablePropertyValues properties) {
052            Collection<HibernateConfigPostProcessor> configPostProcessors = beanFactory.getBeansOfType(HibernateConfigPostProcessor.class).values();
053    
054            PropertyValue property = new PropertyValue("hibernateConfigPostProcessors", configPostProcessors);
055            properties.addPropertyValue(property);
056        }
057    
058        @Override
059        public int getOrder() {
060            return Ordered.HIGHEST_PRECEDENCE;
061        }
062    
063    }
