001    package grails.plugin.hibernatehijacker.indexdsl;
002    
003    import java.util.Iterator;
004    import org.codehaus.groovy.grails.commons.GrailsClassUtils;
005    import org.hibernate.HibernateException;
006    import org.hibernate.cfg.Configuration;
007    import org.hibernate.mapping.Table;
008    import org.slf4j.Logger;
009    import org.slf4j.LoggerFactory;
010    
011    import org.hibernate.mapping.PersistentClass;
012    
013    
014    import grails.plugin.hibernatehijacker.hibernate.HibernateConfigPostProcessor;
015    import groovy.lang.Closure;
016    
017    /**
018     * Configures Hibernate indexes for domain classes
019     * with a static indexes property.
020     * 
021     * @see HibernateIndexBuilder
022     * @author Kim A. Betti
023     */
024    public class IndexDslPostProcessor implements HibernateConfigPostProcessor {
025    
026        private Logger log = LoggerFactory.getLogger(IndexDslPostProcessor.class);
027    
028        @Override
029        @SuppressWarnings("unchecked")
030        public void doPostProcessing(Configuration configuration) throws HibernateException {
031            Iterator<PersistentClass> mappingIterator = configuration.getClassMappings();
032            while (mappingIterator.hasNext()) {
033                PersistentClass persistentClass = mappingIterator.next();
034                if (hasIndexClosure(persistentClass)) {
035                    addIndexesFrom(persistentClass);
036                }
037            }
038        }
039    
040        protected boolean hasIndexClosure(PersistentClass persistentClass) {
041            Class<?> domainClass = persistentClass.getMappedClass();
042            Closure indexes = getIndexClosure(domainClass);
043            return indexes != null;
044        }
045    
046        protected void addIndexesFrom(PersistentClass persistentClass) {
047            Table table = persistentClass.getTable();
048            Closure indexClosure = getIndexClosure(persistentClass.getMappedClass());
049            if (indexClosure != null) {
050                log.info("Reading indexes from " + persistentClass.getClassName());
051                HibernateIndexBuilder.from(table, indexClosure);
052            }
053        }
054    
055        protected Closure getIndexClosure(Class<?> domainClass) {
056            return (Closure) GrailsClassUtils.getStaticPropertyValue(domainClass, "indexes");
057        }
058    
059    }
