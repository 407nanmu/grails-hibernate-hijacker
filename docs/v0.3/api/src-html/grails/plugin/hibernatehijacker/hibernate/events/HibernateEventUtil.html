001    package grails.plugin.hibernatehijacker.hibernate.events;
002    
003    import java.lang.reflect.Array;
004    import java.lang.reflect.Method;
005    
006    import org.hibernate.cfg.Configuration;
007    import org.hibernate.event.EventListeners;
008    
009    
010    
011    /**
012     * 
013     * @see http://www.docjar.com/html/api/org/hibernate/event/EventListeners.java.html
014     * @author Kim A. Betti
015     */
016    public class HibernateEventUtil {
017    
018        public static void addListener(Configuration configuration, String type, Object listener) {
019            EventListeners listeners = configuration.getEventListeners();
020            Class<?> listenerClass = listeners.getListenerClassFor(type);
021            addListener(configuration, type, listenerClass, listener);
022        }
023    
024        @SuppressWarnings("unchecked")
025        private static <T> void addListener(Configuration configuration, String type, Class<T> listenerClass, Object listener) {
026            EventListeners eventListeners = configuration.getEventListeners();
027    
028            try {
029                T[] existingListeners = getExistingListeners(eventListeners, listenerClass);
030                T[] newListeners = (T[]) Array.newInstance(listenerClass, existingListeners.length + 1);
031                System.arraycopy(existingListeners, 0, newListeners, 0, existingListeners.length);
032    
033                newListeners[existingListeners.length] = (T) listener;
034                configuration.setListeners(type, newListeners);
035            } catch (Exception ex) {
036                String message = "Unable to add Hibernate event listener: " + listener + " for type: " + listenerClass.getSimpleName();
037                throw new RuntimeException(message, ex);
038            }
039        }
040    
041        @SuppressWarnings("unchecked")
042        private static <T> T[] getExistingListeners(EventListeners eventListeners, Class<T> type) throws Exception {
043            for (Method method : eventListeners.getClass().getMethods()) {
044                if (method.getName().startsWith("get")) {
045                    Class<?> returnType = method.getReturnType();
046                    if (returnType.isArray() && returnType.getComponentType() == type) {
047                        return (T[]) method.invoke(eventListeners);
048                    }
049                }
050            }
051    
052            throw new Exception("Unable to get existing listeners for " + type.getSimpleName());
053        }
054    
055    }
