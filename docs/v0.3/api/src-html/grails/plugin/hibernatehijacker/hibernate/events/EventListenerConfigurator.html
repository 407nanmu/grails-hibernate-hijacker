001    package grails.plugin.hibernatehijacker.hibernate.events;
002    
003    import grails.plugin.hibernatehijacker.hibernate.HibernateConfigPostProcessor;
004    
005    import org.hibernate.HibernateException;
006    import org.hibernate.cfg.Configuration;
007    
008    /**
009     * Responsible for adding Hibernate event listeners.
010     * 
011     * 
012     * @see http://www.docjar.com/html/api/org/hibernate/event/EventListeners.java.html
013     * @author Kim A. Betti
014     */
015    public class EventListenerConfigurator implements HibernateConfigPostProcessor {
016    
017        /**
018         * create == persist
019         * See the link below for more information on how Hibernate name their event types
020         * @see http://www.docjar.com/html/api/org/hibernate/event/EventListeners.java.html
021         */
022        public static final String EVENT_NAMES = "auto-flush,delete,dirty-check,evict,flush,flush-entity,"
023            + "load,load-collection,lock,merge,create,post-delete,post-insert,post-load,post-update,"
024            + "pre-delete,pre-insert,pre-load,pre-update,refresh,replicate,save-update";
025    
026        private HibernateEventListener hibernateEventListener;
027    
028        @Override
029        public void doPostProcessing(final Configuration configuration) throws HibernateException {
030            registerEventlisteners(configuration);
031        }
032    
033        private void registerEventlisteners(Configuration configuration) {
034            for (String eventName : EVENT_NAMES.split("\\,")) {
035                HibernateEventUtil.addListener(configuration, eventName, hibernateEventListener);
036            }
037        }
038    
039        public void setHibernateEventListener(HibernateEventListener hibernateEventListener) {
040            this.hibernateEventListener = hibernateEventListener;
041        }
042    
043    }
